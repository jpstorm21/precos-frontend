import { I } from '@angular/cdk/keycodes';
import { Component, Inject, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { finalize } from 'rxjs/operators';
import { Privilege } from 'src/app/features/users-management/models/privilege';
import { Profession } from 'src/app/features/users-management/models/profession';
import { Specialization } from 'src/app/features/users-management/models/specialization';
import { UserModel } from 'src/app/features/users-management/models/user.model';
import { UsersService } from 'src/app/features/users-management/services/users/users.service';

interface DialogData {
  title: string,
  user: UserModel,
  professions: Profession[],
  privileges: Privilege[],
  specialization: string[];
}

@Component({
  selector: 'app-edit-user',
  templateUrl: './edit-user.component.html',
  styleUrls: ['./edit-user.component.css']
})
export class EditUserComponent implements OnInit {

  professions: Profession[] = [];
  privileges: Privilege[] = [];
  specializations: Specialization[]=[];

  userForm: FormGroup;
  title: string;
  changePassword: boolean = true;
  isLoading: boolean = true;
  hide:boolean = true;
  password: FormControl;//to save new pasword

  constructor(private diaRef: MatDialogRef<EditUserComponent>, @Inject(MAT_DIALOG_DATA) data: DialogData, private userService: UsersService) {
    this.title = data.title;

    this.userForm = new FormGroup({
      rut: new FormControl(),
      name: new FormControl(),
      profession: new FormControl({ value: null, disabled: this.isLoading }),
      foundation: new FormControl(),
      privilege: new FormControl({ value: null, disabled: this.isLoading }),
      idUser: new FormControl(),
      specialization: new FormControl({ value: null, disabled: this.isLoading }),
      access: new FormControl(true),
    })
    this.password = new FormControl("", [Validators.required])
    if (data.user) {
      this.changePassword = false
      this.userForm.setValue(data.user)

    }
    if (data.professions) {
      this.professions = data.professions;
      this.userForm.get("profession")?.enable();
    }
    if (data.privileges) {
      this.privileges = data.privileges;
      this.userForm.get("privilege")?.enable();
    }
  }

  ngOnInit(): void {
    this.userForm.get("specialization")?.enable();
    let subscriptions: number = 2
    if (!this.professions.length)
      this.userService.getProfessions().pipe(finalize(() => {
        subscriptions--;
        if (subscriptions == 0) {
          this.userForm.get("profession")?.enable();
          this.userForm.get("privilege")?.enable();
          this.isLoading = false;
        }
      })).subscribe(res => {
        this.professions = res.data
      })
    if (!this.privileges.length)
      this.userService.getPrivileges().pipe(finalize(() => {
        subscriptions--;
        if (subscriptions == 0) {
          this.userForm.get("profession")?.enable();
          this.userForm.get("privilege")?.enable();
          this.isLoading = false;
        }
      })).subscribe(res => {
        this.privileges = res.data
      })
    else
      this.isLoading = false;

    this.diaRef.updateSize('450px', 'auto');

  }
  close(): void {
    this.diaRef.close();
  }
  closeEdit(): void {
    this.isLoading = true
    if (this.changePassword && this.password.valid)
      this.userForm.addControl('password', this.password);
    const data = this.userForm.value;
    this.userForm.get("profession")?.disable();
    this.userForm.get("privilege")?.disable();
    this.userForm.get("specialization")?.disable();
    if (!this.userForm.get("idUser")?.value) {
      this.userService.addUser(data).subscribe(res => {
        if (res.data)
          this.userForm.get("idUser")?.setValue(res.data.idUser)
        this.diaRef.close(data);
      });
    }
    else {
      this.userService.editUser(data).subscribe(res => {
        console.log(res.msg);
        this.diaRef.close(data);
      });
    }
  }
}

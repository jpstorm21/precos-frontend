import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { CCRConstants } from 'src/app/core/constants/ccr.constants';
import { CCREnrollmentSurvey } from 'src/app/features/colon-rectal/models/ccr-enrollment-survey';
import { CCRPatient } from 'src/app/features/colon-rectal/models/ccr-patient';
import { CCRRiskSurveyFamily, CCRRiskSurveyFamilyCancer, CCRRiskSurveyGeneral, CCRRiskSurveyHabits, CCRRiskSurveyPathologies } from 'src/app/features/colon-rectal/models/ccr-risk-survey';
import { Coloncheck } from 'src/app/features/colon-rectal/models/coloncheck';
import { Colonoscopy } from 'src/app/features/colon-rectal/models/colonoscopy';
import { CustomHttpResponse } from 'src/app/core/models/http-response';
import { Patient } from 'src/app/features/patient/models/patient';
import { ScreeningSurveyQuestion } from 'src/app/core/models/ScreeningSurveyQuestion';
import { environment } from 'src/environments/environment';

@Injectable({
  providedIn: 'root'
})
export class CCRPatientService {
  private api = environment.ccrPatientsApi;

  constructor(private http: HttpClient) { }

  //enrollment survey
  public registerEnrollmentSurvey(data: CCREnrollmentSurvey): Observable<CustomHttpResponse<any>> {
    return this.http.post<CustomHttpResponse<any>>(this.api + CCRConstants.addEnrollmentSurveyEndpoint, data);
  }

  public updateEnrollmentSurvey(data: CCREnrollmentSurvey): Observable<CustomHttpResponse<any>> {
    return this.http.put<CustomHttpResponse<any>>(this.api + CCRConstants.updateEnrollmentSurveyEndpoint, data);
  }

  public getAllEnrollmentSurveys(): Observable<CustomHttpResponse<CCREnrollmentSurvey[]>> {
    return this.http.get<CustomHttpResponse<CCREnrollmentSurvey[]>>(this.api + CCRConstants.getAllEnrollmentSurveyEndpoint)
  }

  public getPatientEnrollmentSurvey(id: number): Observable<CustomHttpResponse<CCREnrollmentSurvey[]>> {
    return this.http.post<CustomHttpResponse<CCREnrollmentSurvey[]>>(this.api + CCRConstants.getPatientEnrollmentSurveyEndpoint, { idPatient: id });
  }

  //exams
  public addColonCheck(data: Coloncheck): Observable<CustomHttpResponse<any>> {
    return this.http.post<CustomHttpResponse<any>>(this.api + CCRConstants.addColoncheckEndpoint, data)
  }

  public getColoncheckList(id: number): Observable<CustomHttpResponse<Coloncheck[]>> {
    return this.http.post<CustomHttpResponse<Coloncheck[]>>(this.api + CCRConstants.getColoncheckListEndpoint, { idPatient: id })
  }

  public addColonoscopy(data: Colonoscopy): Observable<CustomHttpResponse<any>> {
    return this.http.post<CustomHttpResponse<any>>(this.api + CCRConstants.addColonoscopyEndpoint, data);
  }

  public getColonoscopyList(id: number): Observable<CustomHttpResponse<Colonoscopy[]>> {
    return this.http.post<CustomHttpResponse<Colonoscopy[]>>(this.api + CCRConstants.getColonoscopyListEndpoint, { idPatient: id });
  }

  //CCR Patients
  public getAllCCRPatients(): Observable<CustomHttpResponse<CCRPatient[]>> {
    return this.http.get<CustomHttpResponse<CCRPatient[]>>(this.api + CCRConstants.getAllCCRPatientsEndpoint);
  }

  //patient risk survey
  //general
  public addGeneralRiskSurvey(data:CCRRiskSurveyGeneral):Observable<CustomHttpResponse<any>>{
    return this.http.post<CustomHttpResponse<any>>(this.api+CCRConstants.addRiskSurveyGeneralsEndpoint, data);
  }
  public updateGeneralRiskSurvey(data:CCRRiskSurveyGeneral):Observable<CustomHttpResponse<any>>{
    return this.http.put<CustomHttpResponse<any>>(this.api+CCRConstants.addRiskSurveyGeneralsEndpoint, data);
  }
  public getGeneralRiskSurvey(id:number):Observable<CustomHttpResponse<CCRRiskSurveyGeneral>>{
    return this.http.post<CustomHttpResponse<CCRRiskSurveyGeneral>>(this.api+CCRConstants.addRiskSurveyGeneralsEndpoint, {idPatient:id});
  }

  //pathologies
  public addPathologiesRiskSurvey(data:CCRRiskSurveyPathologies):Observable<CustomHttpResponse<any>>{
    return this.http.post<CustomHttpResponse<any>>(this.api+CCRConstants.addRiskSurveyPathologiesEndpoint, data);
  }
  public updatePathologiesRiskSurvey(data:CCRRiskSurveyPathologies):Observable<CustomHttpResponse<any>>{
    return this.http.put<CustomHttpResponse<any>>(this.api+CCRConstants.updateRiskSurveyPathologiesEndpoint, data);
  }
  public getPathologiesRiskSurvey(id:number):Observable<CustomHttpResponse<CCRRiskSurveyPathologies>>{
    return this.http.post<CustomHttpResponse<CCRRiskSurveyPathologies>>(this.api+CCRConstants.getRiskSurveyPathologiesEndpoint, {idPatient:id});
  }

  //habits
  public addHabitsRiskSurvey(data:CCRRiskSurveyHabits):Observable<CustomHttpResponse<any>>{
    return this.http.post<CustomHttpResponse<any>>(this.api+CCRConstants.addRiskSurveyHabitsEndpoint, data);
  }
  public updateHabitsRiskSurvey(data:CCRRiskSurveyHabits):Observable<CustomHttpResponse<any>>{
    return this.http.put<CustomHttpResponse<any>>(this.api+CCRConstants.updateRiskSurveyHabitsEndpoint, data);
  }
  public getHabitsRiskSurvey(id:number):Observable<CustomHttpResponse<CCRRiskSurveyHabits>>{
    return this.http.post<CustomHttpResponse<CCRRiskSurveyHabits>>(this.api+CCRConstants.getRiskSurveyHabitsEndpoint, {idPatient:id});
  }

  //family background
  public addFamilyBackgroundRiskSurvey(data:CCRRiskSurveyFamily):Observable<CustomHttpResponse<any>>{
    return this.http.post<CustomHttpResponse<any>>(this.api+CCRConstants.addRiskSurveyFamilyBackgroundEndpoint, data);
  }
  public updateFamilyBackgroundRiskSurvey(data:CCRRiskSurveyFamily):Observable<CustomHttpResponse<any>>{
    return this.http.put<CustomHttpResponse<any>>(this.api+CCRConstants.updateRiskSurveyFamilyBackgroundEndpoint, data);
  }
  public getFamilyBackgroundRiskSurvey(data:CCRRiskSurveyFamily):Observable<CustomHttpResponse<CCRRiskSurveyFamily>>{
    return this.http.post<CustomHttpResponse<CCRRiskSurveyFamily>>(this.api+CCRConstants.getRiskSurveyFamilyBackgroundEndpoint, data);
  }

  //family cancer
  public addFamilyCancerRiskSurvey(data:CCRRiskSurveyFamilyCancer):Observable<CustomHttpResponse<any>>{
    return this.http.post<CustomHttpResponse<any>>(this.api+CCRConstants.addRiskSurveyFamilyCancerEndpoint, data);
  }
  public updateFamilyCancerRiskSurvey(data:CCRRiskSurveyFamilyCancer):Observable<CustomHttpResponse<any>>{
    return this.http.put<CustomHttpResponse<any>>(this.api+CCRConstants.updateRiskSurveyFamilyCancerEndpoint, data);
  }
  public getFamilyCancerRiskSurvey(data:CCRRiskSurveyFamilyCancer):Observable<CustomHttpResponse<CCRRiskSurveyFamilyCancer>>{
    return this.http.post<CustomHttpResponse<CCRRiskSurveyFamilyCancer>>(this.api+CCRConstants.getRiskSurveyFamilyCancerEndpoint, data);
  }
}

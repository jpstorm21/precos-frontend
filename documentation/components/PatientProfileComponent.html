import { Component, EventEmitter, Input, OnChanges, OnInit, Output, SimpleChanges, ViewChild } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';
import { MatDialog, MatDialogConfig } from '@angular/material/dialog';
import { MatAccordion } from '@angular/material/expansion';
import { Observable } from 'rxjs';
import { Patient } from 'src/app/features/patient/models/patient';
import { Features, Permission } from 'src/app/features/users-management/models/privilege';
import { Region } from 'src/app/core/models/region';
import { ScreeningSurveyQuestion } from 'src/app/core/models/ScreeningSurveyQuestion';
import { AdministrativeService } from 'src/app/core/services/administrative/administrative.service';
import { PatientService } from 'src/app/features/patient/services/patient.service';
import { ConfirmDialogComponent } from '../../../../shared/components/confirm-dialog/confirm-dialog.component';

@Component({
  selector: 'app-patient-profile',
  templateUrl: './patient-profile.component.html',
  styleUrls: ['./patient-profile.component.css']
})
export class PatientProfileComponent implements OnInit, OnChanges {
  PERMISSIONS = Permission
  FEATURES = Features

  @ViewChild("accordionA") accordionA!: MatAccordion;

  // accordion controls
  isOpenA: boolean = false;
  isOpenB: boolean = false;

  // others
  isReady = false;

  // data mnagement
  @Input() patientId!: number;
  @Input() survey!: ScreeningSurveyQuestion[];
  @Input() enrollmentSurvey: FormGroup;

  @Output() surveyUpdate: EventEmitter<any> = new EventEmitter<any>();

  @Input()isEditable: boolean = false;
  @Output() isEditableChange: EventEmitter<boolean> = new EventEmitter<boolean>();

  regions: Region[] = []
  communes: Region[] = []
  communesLoading: boolean = false;
  patientName: string = 'Desconocido';
  showDeathInfo = false;
  showOptionalRegion = false;
  patientForm: FormGroup;
  constructor(private patientService: PatientService, private adminService: AdministrativeService, private confirm: MatDialog) {
    this.enrollmentSurvey = new FormGroup({});
    this.patientForm = new FormGroup({
      idPatient: new FormControl(),
      name: new FormControl(),
      lastName: new FormControl(),
      lastName2: new FormControl(),
      nationality: new FormControl({ disabled: !this.isEditable }),
      rut: new FormControl,
      birthday: new FormControl(),
      maritalState: new FormControl({ disabled: !this.isEditable }),
      sex: new FormControl(),
      region: new FormControl({ disabled: !this.isEditable }),
      commune: new FormControl({ disabled: !this.isEditable }),
      address: new FormControl(),
      village: new FormControl(),
      residenceTime: new FormControl(),
      previousRegion: new FormControl({ disabled: !this.isEditable }),//check the actual name later
      cesfam: new FormControl({ disabled: !this.isEditable }),
      medicalFacility: new FormControl({ disabled: !this.isEditable }),
      cellphone: new FormControl(),
      emergencyPhone: new FormControl(),
      mail: new FormControl(),
      landline: new FormControl(),
      fonasa: new FormControl({ disabled: !this.isEditable }),
      volunteerAgreement: new FormControl(),
      state: new FormControl({ disabled: !this.isEditable }),
      inactivityReason: new FormControl({ disabled: !this.isEditable }),
      exitDate: new FormControl(),
      cancerDetectionDate: new FormControl(),
      deceased: new FormControl(),
      deceasedByCancer: new FormControl(),
    })
  }
  ngOnInit(): void {
    this.adminService.getRegions().subscribe(res => {
      this.regions = res.data;
    })
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes.patientId) {
      this.patientService.getPatient(this.patientId).subscribe(res => {
        const patientData = res.data[0]
        this.patientName = patientData.name + " " + patientData.lastName + " " + patientData.lastName2;
        this.setPatientData(patientData);
        this.showDeathInfo = this.patientForm.get("deceased")?.value
      })
    }
    if (changes.survey && this.patientId && this.enrollmentSurvey) {
      this.isReady = true;
    }
  }

  public getCommunes(): void {
    this.communesLoading = true;
    this.communes = [];
    this.adminService.getCommuneByRegion(this.patientForm.get('region')?.value).subscribe(res => {
      this.communes = res.data;
      this.communesLoading = false;
    })
  }

  enableEdit(value:boolean):void{
    this.isEditable = value;
    this.isEditableChange.emit(this.isEditable);
  }
  public updateData(): void {
    if (this.patientForm.valid) {
      this.patientService.updatePatient(this.patientForm.value).subscribe(res => {
        console.log(res)
      })
    }
    if (this.enrollmentSurvey.valid) {
      this.surveyUpdate.emit(this.enrollmentSurvey.value);
      this.enableEdit(false);
    }
  }
  showDeceaseDetails(checked: boolean): void {
    this.showDeathInfo = checked;
  }
  // accordions controls
  openAll(): void {
    this.isOpenA = true;
    this.accordionA.openAll();
  }

  closeAll(): void {
    this.isOpenA = false;
    this.accordionA.closeAll();
  }

  closeProfile(): Observable<any> | undefined {
    if (this.isEditable) {
      const config = new MatDialogConfig()
      config.maxWidth="300px"
      config.data = {
        title: "Guardar Cambios",
        msg: "Desea guardar los cambios antes de cerrar? (si desea descartar cambios, recargue la pgina)",
        action: "Guardar y salir"
      }
      return this.confirm.open(ConfirmDialogComponent, config).afterClosed();
    } else
      return undefined;
  }

  public checkOptionalRegion(value: boolean): void {
    if (!value)
      this.showOptionalRegion = true;
    else {
      this.showOptionalRegion = false
      this.patientForm.get("previousRegion")?.reset();
    }
  }
  private setPatientData(patient: Patient): void {
    this.patientForm.setValue({
      idPatient: this.patientId,
      name: patient.name,
      lastName: patient.lastName,
      lastName2: patient.lastName2,
      nationality: patient.nationality,
      rut: patient.rut,
      birthday: patient.birthday,
      maritalState: patient.maritalState,
      sex: patient.sex,
      region: patient.region,
      commune: patient.commune,
      address: patient.adress,
      village: patient.village,
      residenceTime: patient.residenceTime,
      previousRegion: patient.previousRegion,
      cesfam: patient.cesfam,
      medicalFacility: patient.medicalFacility,
      cellphone: patient.cellphone,
      emergencyPhone: patient.emergencyPhone,
      mail: patient.mail,
      landline: patient.landline,
      fonasa: patient.fonasa,
      volunteerAgreement: patient.volunteerAgreement,
      state: patient.state,
      // inactivityReason:patient.inactivityReason
      inactivityReason: "owo",
      // exitDate:patient.exitDate,
      exitDate: new Date(),
      // cancerDetectionDate:patient.cancerDetectionDate
      cancerDetectionDate: new Date(),
      deceased: patient.deceased,
      deceasedByCancer: patient.deceasedByCancer,
    })
  }

}
